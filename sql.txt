create table normal_info(
    normal_no int primary key,
    manage_unit varchar(50),
    maintenance_unit varchar(50),
    build_unit varchar(50),
    design_unit varchar(50),
    supervision_unit varchar(50),
    construction_unit varchar(50),
    built_time date,
    total_cost money,
    type_of_maintenance varchar(10),
    maintenance_level varchar(10),
    road_level varchar(10),
    structure_type varchar(25),
    design_load varchar(25),
    load_limit varchar(25),
    seismic_intensity varchar(25),
    positive_oblique_angle varchar(25),
    bridge_spans int,
    span_combination numeric(12,2),
    bridge_deck_area numeric(12,2),
    total_bridge_length numeric(12,2),
    total_bridge_width numeric(12,2),
    clear_width_of_roadway numeric(12,2),
    clear_width_f_sidewalk numeric(12,2),
    river_grade varchar(10),
    maximum_water_level numeric(10,2),
    constant_water_level numeric(10,2)
)
